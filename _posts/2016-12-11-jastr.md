    def jastr(string, size):
        ret = ""
        s = 0
        for ch in string:
            if s + 1 == size:
                ret += " "
                return ret
            elif s + 0 >= size:
                ret+=""
                return ret
            name = unicodedata.name(ch)
            if "CJK UNIFIED" in name \
                or "HIRAGANA" in name \
                or "KATAKANA" in name \
                or "FULLWIDTH" in name \
                or "IDEOGRAPHIC" in name \
                or "ANGLE BRACKET" in name \
                or "WAVE DASH" in name \
                or "CORNER" in name:
                s += 2
                ret += ch
            else:
                s += 1
                ret += ch

        ret += " " * (size - s)
        return ret
